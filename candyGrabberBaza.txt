-- =========================-- TABELE SA ENUM/STATUSIMA-- =========================CREATE TABLE friendrequeststatus (    id SERIAL PRIMARY KEY,    name VARCHAR(20) NOT NULL UNIQUE);CREATE TABLE itemtype (    id SERIAL PRIMARY KEY,    name VARCHAR(20) NOT NULL UNIQUE);CREATE TABLE powereffect (    id SERIAL PRIMARY KEY,    name VARCHAR(30) NOT NULL UNIQUE);-- =========================-- USERS-- =========================CREATE TABLE users (    id SERIAL PRIMARY KEY,    firstname VARCHAR(50) NOT NULL,    lastname VARCHAR(50) NOT NULL,    username VARCHAR(50) NOT NULL UNIQUE,    password VARCHAR(255) NOT NULL,    email VARCHAR(100) NOT NULL UNIQUE,    profilepicture VARCHAR(255),    gameswon INTEGER DEFAULT 0,    gameslost INTEGER DEFAULT 0);-- =========================-- FRIEND REQUEST-- =========================CREATE TABLE friendrequest (    id SERIAL PRIMARY KEY,    senderid INTEGER NOT NULL,    recipientid INTEGER NOT NULL,    timestamp TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP,    isaccepted VARCHAR(20) NOT NULL,    CONSTRAINT fk_friend_sender        FOREIGN KEY (senderid) REFERENCES users(id) ON DELETE CASCADE,    CONSTRAINT fk_friend_recipient        FOREIGN KEY (recipientid) REFERENCES users(id) ON DELETE CASCADE,    CONSTRAINT fk_friend_status        FOREIGN KEY (isaccepted) REFERENCES friendrequeststatus(name));-- =========================-- MESSAGE-- =========================CREATE TABLE message (    id SERIAL PRIMARY KEY,    senderid INTEGER NOT NULL,    recipientid INTEGER NOT NULL,    content TEXT NOT NULL,    timestamp TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP,    CONSTRAINT fk_message_sender        FOREIGN KEY (senderid) REFERENCES users(id) ON DELETE CASCADE,    CONSTRAINT fk_message_recipient        FOREIGN KEY (recipientid) REFERENCES users(id) ON DELETE CASCADE);-- =========================-- GAME-- =========================CREATE TABLE game (    id SERIAL PRIMARY KEY,    starttime TIMESTAMP WITHOUT TIME ZONE NOT NULL,    duration INTEGER NOT NULL,    status VARCHAR(20) NOT NULL,    winnerid INTEGER,    CONSTRAINT fk_game_winner        FOREIGN KEY (winnerid) REFERENCES users(id));-- =========================-- GAME REQUEST-- =========================CREATE TABLE gamerequest (    id SERIAL PRIMARY KEY,    senderid INTEGER NOT NULL,    recipientid INTEGER NOT NULL,    gameid INTEGER NOT NULL,    timestamp TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP,    isaccepted VARCHAR(20) NOT NULL,    CONSTRAINT fk_gamereq_sender        FOREIGN KEY (senderid) REFERENCES users(id) ON DELETE CASCADE,    CONSTRAINT fk_gamereq_recipient        FOREIGN KEY (recipientid) REFERENCES users(id) ON DELETE CASCADE,    CONSTRAINT fk_gamereq_game        FOREIGN KEY (gameid) REFERENCES game(id) ON DELETE CASCADE,    CONSTRAINT fk_gamereq_status        FOREIGN KEY (isaccepted) REFERENCES friendrequeststatus(name));-- =========================-- ITEM-- =========================CREATE TABLE item (    id SERIAL PRIMARY KEY,    itemtypeid INTEGER NOT NULL,    name VARCHAR(100) NOT NULL,    CONSTRAINT fk_item_type        FOREIGN KEY (itemtypeid) REFERENCES itemtype(id));-- =========================-- CANDY-- =========================CREATE TABLE candy (    itemid INTEGER PRIMARY KEY,    points INTEGER NOT NULL,    CONSTRAINT fk_candy_item        FOREIGN KEY (itemid) REFERENCES item(id) ON DELETE CASCADE);-- =========================-- POWERUP-- =========================CREATE TABLE powerup (    itemid INTEGER PRIMARY KEY,    duration INTEGER NOT NULL,    powereffectid INTEGER NOT NULL,    CONSTRAINT fk_powerup_item        FOREIGN KEY (itemid) REFERENCES item(id) ON DELETE CASCADE,    CONSTRAINT fk_powerup_effect        FOREIGN KEY (powereffectid) REFERENCES powereffect(id));-- =========================-- GAME ITEM-- =========================CREATE TABLE gameitem (    id SERIAL PRIMARY KEY,    gameid INTEGER NOT NULL,    itemid INTEGER NOT NULL,    spawntime TIMESTAMP WITHOUT TIME ZONE NOT NULL,    iscollected BOOLEAN DEFAULT FALSE,    CONSTRAINT fk_gameitem_game        FOREIGN KEY (gameid) REFERENCES game(id) ON DELETE CASCADE,    CONSTRAINT fk_gameitem_item        FOREIGN KEY (itemid) REFERENCES item(id));-- =========================-- PLAYER-- =========================CREATE TABLE player (    id SERIAL PRIMARY KEY,    gameid INTEGER NOT NULL,    userid INTEGER NOT NULL,    CONSTRAINT fk_player_game        FOREIGN KEY (gameid) REFERENCES game(id) ON DELETE CASCADE,    CONSTRAINT fk_player_user        FOREIGN KEY (userid) REFERENCES users(id) ON DELETE CASCADE);-- =========================-- PLAYER ITEM-- =========================CREATE TABLE playeritem (    id SERIAL PRIMARY KEY,    playerid INTEGER NOT NULL,    itemid INTEGER NOT NULL,    quantity INTEGER DEFAULT 1,    isactive BOOLEAN DEFAULT FALSE,    acquiredat TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP,    CONSTRAINT fk_playeritem_player        FOREIGN KEY (playerid) REFERENCES player(id) ON DELETE CASCADE,    CONSTRAINT fk_playeritem_item        FOREIGN KEY (itemid) REFERENCES item(id));
